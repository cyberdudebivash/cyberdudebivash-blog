# SENTINEL APEX API — Python Example
# pip install requests

import requests

API_KEY = "sa_YOUR_API_KEY_HERE"
BASE_URL = "https://intel.cyberdudebivash.com/api/intel"

# Fetch latest critical CVEs (CVSS >= 9.0)
response = requests.get(
    f"{BASE_URL}/cve/latest",
    headers={"X-API-Key": API_KEY},
    params={"limit": 5, "min_cvss": 9.0, "kev_only": "true"}
)

data = response.json()

for cve in data["vulnerabilities"]:
    print(f"[{cve['severity']}] {cve['id']} — CVSS {cve['cvss_score']}")
    print(f"  {cve['description'][:80]}...")
    print(f"  KEV: {cve['cisa_kev']} | Exploited: {cve['actively_exploited']}")
    print()
